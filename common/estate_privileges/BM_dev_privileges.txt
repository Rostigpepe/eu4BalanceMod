#####		FOREST EXPANSION		#####
estate_burghers_forest_expansion = {
	icon = privilege_dev_cost_forest
	max_absolutism = -10
	influence = 0.15
	loyalty = 0.1
	is_valid = {
		OR = {
			has_unlocked_estate_privilege = { estate_privilege = estate_burghers_forest_expansion }
			OR = {
				AND = {
					is_year = 1444
				}
				# RUSSIA/RUSSIAN MINORS
				tag = RUS
				tag = MOS
				tag = KRL
				tag = NOV
				tag = PSK
				tag = TVE
				tag = YAR
				tag = RSO
				tag = NZH
				tag = OKA
				tag = RYA
				tag = PRM
				tag = BLO
				# BALTIC + LITHUANIA
				tag = SMO
				tag = PLT
				tag = LIT
				tag = LIV
				tag = LVA
				tag = KUR
				tag = EST
				tag = LTG
				tag = CHR
				# SAMI + KARELIA
				tag = KRL
				tag = SMI
				# COLONIAL NATIONS
				tag = QUE
				tag = CAN
				tag = USA
			}
		}
	}
	can_select = {
		any_owned_province = {
			OR = {
				has_terrain = forest
				has_terrain = woods
			}
		}
	}
	on_granted = {
		custom_tooltip = estate_burghers_forest_expansion_gt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_terrain = forest
				}
				add_province_modifier = {
					name = estate_burghers_forest_expansion_forest
					duration = -1
				}
			}
			every_owned_province = {
				limit = {
					has_terrain = woods
				}
				add_province_modifier = {
					name = estate_burghers_forest_expansion_woods
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_burghers_forest_expansion_rt
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = estate_burghers_forest_expansion_forest
						has_province_modifier = estate_burghers_forest_expansion_woods
					}
				}
				remove_province_modifier = estate_burghers_forest_expansion_forest
				remove_province_modifier = estate_burghers_forest_expansion_woods
			}
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 2
			num_of_owned_provinces_with = {
				OR = {
					has_terrain = woods
					has_terrain = forest
				}
				value = 8
			}
		}
		modifier = {
			factor = 0
			is_absolutism_active = yes
		}
	}
}
