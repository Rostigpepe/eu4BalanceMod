country_event = {
	id = ME_Muscovy_Events.4
	title = ME_Muscovy_Events.4.t
	desc = ME_Muscovy_Events.4.d
	picture = OVEREXTENSION_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			had_country_flag = {
				flag = muscovy_noble_lands
				days = 3650 #10 years
			}
			had_country_flag = {
				flag = muscovy_crown_lands
				days = 3650 #10 years
			}
		}
	}

	after = {
		clr_country_flag = muscovy_noble_lands
		clr_country_flag = muscovy_crown_lands
	}

	option = {
		name = ME_Muscovy_Events.4.a

		if = {
			limit = { has_country_flag = muscovy_noble_lands }
			add_yearly_manpower = 0.1
			custom_tooltip = ME_Muscovy_Events.4.tta
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_noble_lands }
					remove_province_modifier = muscovy_noble_lands
					set_local_autonomy = 10
				}
			}
		}
		else = {
			add_years_of_income = 0.083
			custom_tooltip = ME_Muscovy_Events.4.ttb
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_crown_lands }
					remove_province_modifier = muscovy_crown_lands
					set_local_autonomy = 25
				}
			}
		}

		add_adm_power = -100
		add_stability = -1
	}

	option = {
		name = ME_Muscovy_Events.4.b

		if = {
			limit = { has_country_flag = muscovy_noble_lands }
			custom_tooltip = ME_Muscovy_Events.4.tta
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_noble_lands }
					remove_province_modifier = muscovy_noble_lands
					set_local_autonomy = 10
				}
			}
		}
		else = {
			custom_tooltip = ME_Muscovy_Events.4.ttb
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_crown_lands }
					remove_province_modifier = muscovy_crown_lands
					set_local_autonomy = 25
				}
			}
		}

		add_prestige = -25
		add_accepted_culture_or_dip_power = { culture = novgorodian }
	}

	option = {
		name = ME_Muscovy_Events.4.c

		add_prestige = -5
	}
}
