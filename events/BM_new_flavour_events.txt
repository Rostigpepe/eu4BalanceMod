#Subjugation of Novgorod and subsequent coring

country_event = {
	id = BM_muscovy_conquest_of_novgorod.1
	title = "BM_muscovy_conquest_of_novgorod.1.t"
	desc = "BM_muscovy_conquest_of_novgorod.1.d"
	picture = CONQUEST_eventPicture

	fire_only_once = yes
	trigger = {
		OR = {
			AND = {
				war_with = NOV
				war_score_against = {
					who = NOV
					value = 75
				}
				controls = 310
			}
		}
	}

	option = {
		name = BM_muscovy_conquest_of_novgorod.1.A
		NOV = {
			country_event = {
				id = BM_peace_offer_to_novgorod.1 #A Merciful Offer
				days = 1
			}
		}
	}
	option = {
		name = BM_muscovy_conquest_of_novgorod.1.B
		add_country_modifier = {
			name = muscovy_muscovite_supremacy
			duration = -1 #until the end of the war
			desc = muscovy_muscovite_supremacy_tt
		}
	}
}

country_event = {
	id = BM_muscovite_consolidation_of_novgorod_lands.1
	title = BM_muscovite_consolidation_of_novgorod_lands.1.t
	desc = BM_muscovite_consolidation_of_novgorod_lands.1.d
	picture = DEBATE_REPUBLICAN_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = BM_muscovite_consolidation_of_novgorod_lands.1.A

		custom_tooltip = BM_muscovite_consolidation_of_novgorod_lands.1.ATooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					NOT = { is_core = FROM }
				}
				add_unrest = -2
			}
		}

		custom_tooltip = BM_muscovite_consolidation_of_novgorod_lands.1.BToolTip
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = FROM
					NOT = { is_core = ROOT }
				}
				add_territorial_core = ROOT

				set_local_autonomy = 40
				add_province_modifier = {
					name = muscovy_noble_lands
					duration = 9125 #25 years
				}
			}
		}

		set_country_flag = muscovy_noble_lands

		change_estate_land_share = {
			estate = estate_nobles
			share = 5
		}
	}

	option = {
		name = BM_muscovite_consolidation_of_novgorod_lands.1.B

		custom_tooltip = BM_muscovite_consolidation_of_novgorod_lands.1.ATooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					NOT = { is_core = FROM }
				}
				add_unrest = 1
				add_adm_power = 100
			}
		}

		custom_tooltip = BM_muscovite_consolidation_of_novgorod_lands.1.CTooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = FROM
					NOT = { is_core = ROOT }
				}
				add_territorial_core = ROOT

				set_local_autonomy = 20
				add_province_modifier = {
					name = muscovy_crown_lands
					duration = 9125 #25 years
				}
			}
		}

		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -10
		}

		set_country_flag = muscovy_crown_lands

		310 = { add_core = ROOT }
		33 = { add_core = ROOT }
		1955 = { add_core = ROOT }

		add_adm_power = -100

		change_estate_land_share = {
			estate = all
			share = -10
		}
	}
}

country_event = {
	id = BM_the_governing_of_novgorod.1
	title = BM_the_governing_of_novgorod.1.t
	desc = BM_the_governing_of_novgorod.1.d
	picture = OVEREXTENSION_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			had_country_flag = {
				flag = muscovy_noble_lands
				days = 3650 #10 years
			}
			had_country_flag = {
				flag = muscovy_crown_lands
				days = 3650 #10 years
			}
		}
	}

	after = {
		clr_country_flag = muscovy_noble_lands
		clr_country_flag = muscovy_crown_lands
	}

	option = {
		name = BM_the_governing_of_novgorod.1.A

		if = {
			limit = { has_country_flag = muscovy_noble_lands }
			add_yearly_manpower = 0.1
			custom_tooltip = BM_the_governing_of_novgorod.1.ATooltip
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_noble_lands }
					remove_province_modifier = muscovy_noble_lands
					set_local_autonomy = 10
				}
			}
		}
		else = {
			add_years_of_income = 0.5
			custom_tooltip = BM_the_governing_of_novgorod.1.BTooltip
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_crown_lands }
					remove_province_modifier = muscovy_crown_lands
					set_local_autonomy = 20
				}
			}
		}

		add_stability = -1
	}

	option = {
		name = BM_the_governing_of_novgorod.1.B

		if = {
			limit = { has_country_flag = muscovy_noble_lands }
			custom_tooltip = BM_the_governing_of_novgorod.1.ATooltip
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_noble_lands }
					remove_province_modifier = muscovy_noble_lands
					set_local_autonomy = 10
				}
			}
		}
		else = {
			custom_tooltip = BM_the_governing_of_novgorod.1.BTooltip
			hidden_effect = {
				every_owned_province = {
					limit = { has_province_modifier = muscovy_crown_lands }
					remove_province_modifier = muscovy_crown_lands
					set_local_autonomy = 20
				}
			}
		}

		add_stability = 1
		add_accepted_culture_or_dip_power = { culture = novgorodian }
	}

	option = {
		name = BM_the_governing_of_novgorod.1.C

		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
	}
}
