country_event = {
	id = BM_starting_buffs_eng.2
	title = BM_starting_buffs_eng.2.t
	desc = BM_starting_buffs_eng.2.d
	picture = LAND_MILITARY_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	trigger = {
		AND = {
			tag = ENG
			is_at_war = no
		}
	}
	mean_time_to_happen = {
		days = 10
	}
	option = {
		name = BM_starting_buffs_eng.2.a #We must be on the defensive, build new Castles and reinforce the old!
		1879 = {
			add_building_construction = {
				building = fort_15th
				speed = 0.5
				cost = 0.25
			}
			add_province_modifier = {
				name = BM_modern_english_castles
				duration = 7300
			}
		}
		174 = {
			add_building_construction = {
				building = fort_15th
				speed = 0.5
				cost = 0.25
			}
			add_province_modifier = {
				name = BM_modern_english_castles
				duration = 7300
			}
		}
		every_owned_province = {
			limit = {
				region = france_region
				has_building = fort_15th
			}
			add_province_modifier = {
				name = BM_modern_english_castles
				duration = 7300
			}
		}
	}
	option = {
		name = BM_starting_buffs_eng.2.b #Ready the Longbows for another Agincourt, we are going on the offensive!
		add_yearly_manpower = 1
		add_country_modifier = {
			name = BM_readying_the_longbows
			duration = 7300
		}
		set_country_flag = BM_ready_longbows_flag
	}
	option = {
		name = BM_starting_buffs_eng.2.c #The French are splintered between rivaling Appanages, let's sow discontent!
		add_treasury = -400
		FRA = {
			every_subject_country = { 
				add_liberty_desire = 15
			}
		}
	}
	option = {
		name = BM_starting_buffs_eng.2.d #War? But that would ruin my dev-cost modifiers!
		add_prestige = -1
		add_ruler_personality = naive_personality
	}
}
